      \begin{block}{Motivation}

        \textbf{1-WL bounds plain GNNs} \citep{xu_how_2019}. Topological
        lifts escape it by message-passing over a richer object:
      \begin{itemize}\setlength{\itemsep}{0.15em}
          \item \textbf{MPSN} \citep{bodnar_weisfeiler_2021}: biased to \emph{cliques}.
          \item \textbf{CWN} \citep{bodnar_weisfeiler_2022}: biased to \emph{induced cycles}.
          \item \textbf{PCN} \citep{truong_weisfeiler_2024}: simple paths, no substructure bias.
        \end{itemize}

        \medskip
        \textbf{PCN leaves two problems on the table:}
      \begin{enumerate}\setlength{\itemsep}{0.15em}
          \item Symmetric scatter-sum init forces
            $A{\to}B{\to}C$ and $C{\to}B{\to}A$ to receive \emph{identical}
            embeddings.
          \item Lift is monolithic: paths \emph{or} rings.
            Link prediction needs the closed-cycle signal that paths
            alone discard.
        \end{enumerate}

      \end{block}

      \medskip

      \begin{alertblock}{Our Contributions}

      \begin{enumerate}\setlength{\itemsep}{0.2em}
          \item \textbf{Position-aware learnable init.} $k{+}1$ position-specific
            linear maps per $k$-cell: strictly subsumes the scatter-sum
            and breaks orientation symmetry at $t{=}0$.
          \item \textbf{Unified path and cellular complex.} Path- and ring-cells
            coexist at dim~2; canonicalised edges route messages between
            them for free.
          \item \textbf{Theory (U-PWL).} U-PWL\,$\geq$\,PWL \emph{and} U-PWL\,$\geq$\,CWL(\textit{k}-IC); strictly $>$ 1-WL.
          \item \textbf{Cora ablation, 10 seeds.} Each modification cuts seed
            variance; unified lift achieves the best mean test AUC.
        \end{enumerate}

      \end{alertblock}

      \medskip

      \begin{exampleblock}{Big Takeaway}

        \large
        \textbf{Unified path- and ring-cells with a learnable,
          order-aware init give PCN strictly greater expressivity and the
        best Cora AUC --- at \emph{zero} kernel-level cost.}

        \medskip\normalsize
      \begin{itemize}\setlength{\itemsep}{0.15em}
          \item \textbf{Greater expressivity:} proved via U-PWL theory.
          \item \textbf{Lower variance:} $-31\%$ test-AUC std across 10 seeds.
          \item \textbf{Zero kernel cost:} same MP architecture as PCN.
        \end{itemize}

      \end{exampleblock}

      \medskip

      \begin{block}{Background: Path Complexes}

        A \emph{simple $k$-path} is an ordered tuple
        $\sigma=(v_0,\ldots,v_k)$ of distinct vertices with
        $\{v_i,v_{i+1}\}\!\in\!\mathcal E$; write $P_k(G)$ for the
        full set. The boundary alternates end-deletion,
        $\partial\,e_{i_0\ldots i_p}=\sum_{q}(-1)^q\,e_{i_0\ldots\hat{i}_q\ldots i_p}$
        \citep{grigoryan_homologies_2013}, and PCN message-passes over
        the faces $\mathcal B(\sigma)$ together with the upper
        adjacencies $\mathcal N_\uparrow(\sigma)$.

        \medskip
        \centering
        \includegraphics[width=0.55\colwidth]{../paper/path_complex_illustration.pdf}
        {\footnotesize\\$\mathcal S_2\subset\mathcal A_2$: path complexes
        strictly enrich the 2-cell vocabulary.}

      \end{block}

      \medskip

      \begin{block}{The Deterministic-Init Problem}

        PCN initialises cell features by a symmetric aggregation,
        $h_\sigma^{(0)}=\bigoplus_{v\in\sigma}x_v$ with
        $\oplus\!\in\!\{\text{sum},\text{mean}\}$
        \citep{truong_weisfeiler_2024}, which immediately collapses
        orientation: $h_{(A,B,C)}^{(0)}=h_{(C,B,A)}^{(0)}$. The damage
        is permanent, since later permutation-invariant aggregation also
        discards the signed boundary, so the source/target asymmetry
        that link decoders rely on is erased before message passing
        even starts.

      \end{block}

    \end{column}

    \begin{column}{\colwidth}

      \begin{block}{Notation}

        \begin{tabular}{@{}l@{\hspace{0.9em}}l@{}}
          $P_k(G),R_L(G)$ & $k$-paths;\,induced cycles \\
          $\sigma=(v_0,\ldots,v_k)$ & ordered $k$-cell \\
          $\mathcal B,\mathcal N_\uparrow$ & faces; upper-adjacents \\
          $W_i^{(k)}$ & pos.-aware init weight \\
          $\mathcal C(G)$ & unified path and ring complex \\
        \end{tabular}

      \end{block}

      \medskip

      \begin{block}{Method 1: Position-Aware Learnable Init}

        \textbf{Replace} the scatter-sum with $k{+}1$ position-specific maps:
        \[
          \boxed{\;
            h_\sigma^{(0)} \;=\; \sum_{i=0}^{k} W_i^{(k)}\,x_{v_i}
          \;}
        \]
        Each node contributes once per cell, so the asymptotic cost is
        unchanged. Sharing $W_i^{(k)}=W$ recovers a linear projection
        of the scatter-sum, so the new init strictly subsumes the
        deterministic one; for unequal weights, paths $\sigma\neq\sigma'$
        with the same multi-set of endpoints receive \emph{distinct}
        features, something no symmetric $\bigoplus$ can produce. The
        first vs.\ last endpoint encodes source vs.\ target, exactly
        the asymmetry link decoders already exploit. The price is
        $(k{+}1)\,d\,d_{\!\textrm{in}}$ parameters per $k$-cell ($d{=}64$
        here); the init still runs in $\mathcal O(|P_k|\,d)$.

      \end{block}

      \medskip

      \begin{block}{Method 2: Unified Path and Cellular Complex}

        Stack ring-cells \emph{on top of} the path complex at dim~2:
        \[
          \mathcal C_2(G)\;=\;\underbrace{P_2(G)}_{\text{2-paths}}\;\cup\;
          \underbrace{R_L(G)}_{\text{induced cycles},\,L\leq 7}
        \]
        Path- and ring-cells have different boundary rules but share a
        single edge representation: a ring and a path on the same edge
        resolve to the \emph{same} canonical edge cell, so
        upper-adjacency mixes ring\,$\leftrightarrow$\,path messages
        for free.

        \heading{Why this targets link prediction}

        Common neighbours and short cycles drive link prediction
        (Adamic--Adar, SEAL). Path-cells expose only the \emph{open}
        neighbourhoods of $(u,v)$, whereas ring-cells expose the
        \emph{closed} ones, supplying exactly the signal that
        path-only PCN misses.

        \medskip
        \heading{Boundary rules differ; adjacency unifies}

        The path boundary is the signed alternating end-deletion
        $\partial(v_0,\!\ldots\!,v_k)\!=\!\sum_q(-1)^q(v_0,\!\ldots\!,\hat v_q,\!\ldots\!,v_k)$,
        while the ring boundary collects the cycle's edges unsigned
        (the standard cellular $\partial_2$). Both reduce to the
        \emph{same} canonical edge IDs, so a single
        $\mathcal N_\uparrow$ graph carries both kinds of message
        with no extra MP kernel. On Cora this adds $+26\%$ dim-2
        cells; each triangle contributes $1$ ring and $3$ paths, so
        rings supply the triadic-closure signal that paths alone
        cannot encode.

      \end{block}

      \medskip

      \begin{block}{Construction Recipe}

      \begin{enumerate}\setlength{\itemsep}{0.18em}
          \item Enumerate $P_2(G)$ and $R_{\leq 7}(G)$ separately
            (one-off preprocessing).
          \item Canonicalise every edge by sorting endpoint pairs; this is
            the \emph{single} representation any 1-cell receives.
          \item Build boundary and upper-adjacency tensors over the
            merged dim-2 set $\mathcal C_2(G)=P_2\cup R_{\leq 7}$.
          \item Run standard PCN message passing; the position-aware
            init keeps orientation through every layer.
        \end{enumerate}

      \end{block}

      \begin{block}{Worked Example: Symmetry Breaking}

        Take graph $G$: $a{-}b{-}c$ with one-hot features
        $x_a{=}e_1,\,x_b{=}e_2,\,x_c{=}e_3$.
        Compare reversed paths $\sigma{=}(a,b,c)$ and $\sigma'{=}(c,b,a)$.

        \medskip
        \textbf{Deterministic init} collapses orientation:
        \[
          h_\sigma^{(0)} \;=\; e_1{+}e_2{+}e_3 \;=\; h_{\sigma'}^{(0)}.
        \]

        \medskip
        \textbf{Learnable init} with
        $W_0^{(2)}{=}I,\;W_1^{(2)}{=}\mathbf 0,\;W_2^{(2)}{=}2I$:
        \begin{align*}
          h_\sigma^{(0)}    &= I\,e_1 + \mathbf 0\,e_2 + 2I\,e_3
          \;=\; e_1 + 2\,e_3,\\
          h_{\sigma'}^{(0)} &= I\,e_3 + \mathbf 0\,e_2 + 2I\,e_1
          \;=\; e_3 + 2\,e_1\;\neq\;h_\sigma^{(0)}.
        \end{align*}

        \medskip
        Orientation therefore \textbf{survives} initialization and
        flows through every later message-passing step.

      \end{block}

      \begin{block}{Message Passing on the Unified Complex}

        Each layer aggregates boundaries and upper-adjacent cells:
        {\small
          \begin{align*}
            m_\mathcal B(\sigma) &= \sum_{\tau\in\mathcal B(\sigma)} M_\mathcal B(h_\tau^{(t)}),\;\;
            m_\uparrow(\sigma) = \sum_{\tau\in\mathcal N_\uparrow(\sigma)} M_\uparrow(h_\tau^{(t)}), \\[2pt]
            h_\sigma^{(t+1)} &= \mathrm{UP}\bigl(h_\sigma^{(t)},\,m_\mathcal B(\sigma),\,m_\uparrow(\sigma)\bigr).
          \end{align*}
        }%
        $M_\mathcal B,\,M_\uparrow,\,\mathrm{UP}$: GIN-style MLPs \citep{xu_how_2019}.

        \medskip
        \textbf{Drop-in.} The same sums silently mix
        \textcolor{orange!75!black}{\textbf{ring}} and \textcolor{blue!70!black}{\textbf{path}}
        messages with the \emph{kernel unchanged}.

      \end{block}

      \medskip

      \begin{exampleblock}{Theoretical Results}

        \heading{U-PWL: WL test on $\mathcal C(G)$}
        \[
          c_\sigma^{(t+1)}=\textsc{Hash}\!\bigl(
            c_\sigma^{(t)},\;
            \{\!\{c_\tau^{(t)}\}\!\}_{\mathcal B(\sigma)},\;
            \{\!\{c_\tau^{(t)}\}\!\}_{\mathcal N_\uparrow(\sigma)}
          \bigr)
        \]

        \medskip
      \begin{itemize}\setlength{\itemsep}{0.15em}
          \item \textbf{Thm 1.} U-PWL $\geq$ PWL.
          \item \textbf{Thm 2.} U-PWL $\geq$ CWL(\textit{k}-IC) at dim~2.
          \item \textbf{Cor.}   U-PWL strictly $>$ 1-WL; not less than 3-WL.
        \end{itemize}

        \medskip
        \emph{Why:} adding ring-cells preserves every PCN path-cell and
        adjacency, so U-PWL refines both PWL and CWL(\textit{k}-IC)
        \citep{bodnar_weisfeiler_2021,bodnar_weisfeiler_2022,truong_weisfeiler_2024}.

      \end{exampleblock}

        \textbf{Limitations.} Cora gains are modest ($<\!1\%$ AUC);
        stacked modifications interact; closing the gap to SEAL /
        NBFNet requires larger benchmarks.

        \textbf{Future.}
      \begin{itemize}\setlength{\itemsep}{0.15em}

            (each 3-cycle is one ring and three paths).
